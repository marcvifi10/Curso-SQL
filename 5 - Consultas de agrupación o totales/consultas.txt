SELECT SECCION, AVG(PRECIO) AS MEDIA FROM PRODUCTOS GROUP BY SECCION ORDER BY MEDIA

SELECT POBLACION, COUNT(CODIGO) AS N_CLIENTES FROM CLIENTES GROUP BY POBLACION
