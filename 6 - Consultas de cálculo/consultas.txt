SELECT NOMBRE, SECCIO, PRECIO, ROUND(PRECIO*1.21,2) AS PRECIO_CON_IVA FROM PRODUCTOS

SELECT NOMBRE, SECCIO, PRECIO, ROUND(PRECIO-3,2) AS PRECIO_CON_DESCUENTO FROM PRODUCTOS

SELECT NOMBRE, SECCION, PRECIO, FECHA, NOW() AS FECHA_ACTUAL, DATEDIFF(NOW(),FECHA) FROM PRODUCTOS

SELECT NOMBRE, SECCION, PRECIO, FECHA, DATE_FORMAT(NOW(),"%D-%M") AS FECHA_ACTUAL, DATEDIFF(NOW(),FECHA) FROM PRODUCTOS
